[Sciter] 实现平滑过渡的进度条

2021-07-27

在处理一些任务的时候，进度不总是可以量化的，一个假的进度条实现起来会更方便。

功能：进度每秒加1，到99时会终止。且中途可以设置进度到某个特定的值，这里的特点是进度要平滑地过度。

方法一

模拟实现sleep函数，然后在循环里累加。

<html>
<script>
    let iid;
    let newPercent = 0;

    function sleep(time) {
        return new Promise(resolve => setTimeout(resolve, time));
    }

    async function doTick() {
        const el = document.querySelector('progress');
        let current = el.value;
        const max = el.getAttribute('max') - 1;
        if (current < max) {
            if (current < newPercent) {
                while (current < newPercent) {
                    await sleep(10);
                    el.value = ++current;
                }
            } else {
                el.value = ++current;
            }
        } else {
            clearInterval(iid);
        }
    }

    document.querySelector('button').onclick = () => {
        newPercent = 70;
    }

    document.ready = () => {
        iid = setInterval(doTick, 1000);
    }
</script>

<body>
    <progress max="100" value="0"></progress>
    <button>test</button>
</body>

</html>

方法二

上面的方法有点笨，更好的方法是通过元素的 animate 方法来实现。

<html>
<script>
    let iid;

    function doTick() {
        const el = document.querySelector('progress');
        const max = el.getAttribute('max') - 1;
        if (el.value < max) {
            el.value++;
        } else {
            return true;
        }
    }

    document.querySelector('button').onclick = () => {
        const newValue = 90;
        const el = document.querySelector('progress');
        clearInterval(iid);

        const left = newValue - el.value;
        const old = el.value;
        el.animate(function (progress) {
            if (progress > 0) {
                this.value = old + left * progress;
                if (progress == 1) {
                    iid = setInterval(doTick, 1000);
                }
            }
            return true;
        }, { duration: 500 });
    }

    document.ready = () => {
        iid = setInterval(doTick, 1000);
    }
</script>

<body>
    <progress max="100" value="10"></progress>
    <button>go</button>
</body>

</html>
